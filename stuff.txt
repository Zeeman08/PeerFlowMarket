CREATE TABLE "STOREFRONT" (
  "STOREFRONT_ID" SERIAL,
  "DESCRIPTION" TEXT,
  "RATING" REAL,
  "LAST_UPDATED_ON" TIMESTAMP,
  "IMAGE" TEXT,
  PRIMARY KEY ("STOREFRONT_ID")
);

CREATE INDEX "Key" ON  "STOREFRONT" ("RATING", "LAST_UPDATED_ON", "IMAGE");

CREATE TABLE PRODUCT (
  PRODUCT_ID SERIAL PRIMARY KEY,
  PRODUCT_NAME TEXT,
  PRODUCT_DESCRIPTION TEXT,
  PRICE REAL,
  IMAGE TEXT,
  RATING REAL,
  STOREFRONT_ID SERIAL,
  TAGS TEXT,
  CONSTRAINT FK_PRODUCT_STOREFRONT_ID
    FOREIGN KEY (STOREFRONT_ID)
      REFERENCES STOREFRONT(STOREFRONT_ID)
);
CREATE INDEX Key ON PRODUCT (PRODUCT_NAME, PRICE, IMAGE, RATING, TAGS);	

CREATE TABLE ANNOUNCEMENTS (
	ANNOUNCEMENT_ID SERIAL PRIMARY KEY,
	STOREFRONT_ID SERIAL,
	ANNOUNCEMENT_DESCRIPTION TEXT,
	POSTED_ON TIMESTAMP,
  	IMAGE TEXT,
  CONSTRAINT FK_ANNOUNCEMENT_STOREFRONT_ID
    FOREIGN KEY (STOREFRONT_ID)
      REFERENCES STOREFRONT(STOREFRONT_ID)
);
CREATE INDEX Key_announcement ON ANNOUNCEMENTS (POSTED_ON);
